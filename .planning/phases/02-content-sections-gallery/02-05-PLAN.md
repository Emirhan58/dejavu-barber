```javascript
// scripts/generate-blur.mjs
import sharp from 'sharp';
import { readFileSync } from 'fs';
import path from 'path';

const images = [
  'public/images/hero/dejavu-kuafor.jpg',
  'public/images/about/kuafor-icerisi.jpg',
  'public/images/gallery/dejavu-kuafor.jpg',
  'public/images/gallery/dejavu-kuafor2.jpg',
  'public/images/gallery/kuafor-icerisi.jpg',
  'public/images/gallery/kuafor-ici.jpg',
  'public/images/gallery/kuafor-ici2.jpg',
  'public/images/gallery/sac.jpg',
  'public/images/gallery/sac2.jpg',
];

for (const img of images) {
  const buffer = await sharp(img).resize(8, 8).jpeg({ quality: 40 }).toBuffer();
  const base64 = `data:image/jpeg;base64,${buffer.toString('base64')}`;
  console.log(`${path.basename(img)}: ${base64}`);
}
```
